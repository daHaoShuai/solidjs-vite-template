### solidjs + @solidjs/router + vite + unocss
```
{
  "devDependencies": {
    "unocss": "^0.45.29",
    "vite": "^3.1.0"
  },
  "dependencies": {
    "@solidjs/router": "^0.5.0",
    "solid-js": "^1.5.7",
    "vite-plugin-solid": "^2.3.9"
  }
}
```

### 使用和打包
```
# 安装依赖
pnpm i
# 允许项目
pnpm dev
# 打包
pnpm build
# 预览打包完成的项目
pnpm preview
```
